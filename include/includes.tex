% Points table
\newcommand{\points}[2]{
    \newcounter{n}
    \setcounter{n}{#2}
    \stepcounter{n}
    \newcounter{ct}
    \begin{flushleft}
        \Large{
            \begin{tabular}{*{\value{n}}{|p{0.08\textwidth}}|}
                \hline
                \forloop{ct}{#1}{\value{ct} < \value{n}}{\arabic{ct}&}
                $\Sigma$\\
                \hline
                \forloop{ct}{#1}{\value{ct} < \value{n}}{&}\\
                \hline
            \end{tabular}
        }
    \end{flushleft}
}
